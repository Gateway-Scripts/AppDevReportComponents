public string StructureId { get; set; }
public string Volume { get; set; }
public string Type { get; set; }
public bool HighRes { get; set; }
public string MinDose { get; set; }
public string MeanDose { get; set; }
public string MaxDose { get; set; }

public StructurePropertyModel(Structure structure, PlanningItem plan)
{
    SetValues(structure, plan);
}
private void SetValues(Structure structure, PlanningItem plan)
{
    StructureId = structure.Id;
    Volume = structure.Volume.ToString("F1");
    Type = structure.DicomType;
    HighRes = structure.IsHighResolution;
    if (structure.IsEmpty)
    {
        MinDose = MeanDose = MaxDose = "-";
    }
    else
    {
        //get dvh for structure.
        DVHData dvh = plan.GetDVHCumulativeData(structure, VMS.TPS.Common.Model.Types.DoseValuePresentation.Absolute,
            VMS.TPS.Common.Model.Types.VolumePresentation.Relative, 1);
        if (dvh == null)
        {
            MinDose = MeanDose = MaxDose = "-";
        }
        else
        {
            MinDose = dvh.MinDose.ToString();
            MeanDose = dvh.MeanDose.ToString();
            MaxDose = dvh.MaxDose.ToString();
        }
    }
}
